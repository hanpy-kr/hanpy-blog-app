export const postDetectionPreprocessing = async (info: any) => {
  // set parameter
  let formData = new FormData()
  formData.append(
    'file',
    await convertFileUrlToFileObjectWithFile(
      info.uploadSection.file,
      info.fileName,
      info.uploadSection.type,
    ),
  )

  formData.append('info', JSON.stringify(info))

  const response = await detectionFormDataApi.post<any>(
    '/pre-processing',
    formData,
  )

  return checkResponse(response.data)
}
